	  if (request.status >= 200 && request.status < 400) {
		// Success!

	  }
	};

	request.onerror = function() {
	  // There was a connection error of some sort

Restful API:以resource為中心（CRUD = create/read/update/delete）。
	GET/posts		列出資源
	POST/posts		新增資源
	GET/posts/id	得到資源
	PUT/posts/id	修改資源
	DELETE/posts/id	刪除資源
